kill_prime_minister = {
	is_shown = {
		has_variable = debug_enabled
	}
	possible = {
	}
	when_taken = {
		var:prime_minister = {
			kill_character = yes
			remove_variable = prime_minister_trigger
		}
		remove_variable = prime_minister
	}
	ai_chance = {
		base = 0
	}
}

replace_prime_minister = {
	is_shown = {
		has_variable = debug_enabled
	}
	possible = {
	}
	when_taken = {
		var:prime_minister = {
			kill_character = yes
			remove_variable = prime_minister_trigger
		}
		create_character = {
			on_created = {
				set_variable = prime_minister_trigger
				owner = {
					set_variable = {
						name = prime_minister
						value = PREV
					}
				}
			}		
		}
	}
	ai_chance = {
		base = 0
	}
}

elect_prime_minister = {
	is_shown = {
		has_variable = debug_enabled
	}
	possible = {
	}
	when_taken = {
		var:prime_minister = {
			remove_variable = prime_minister_trigger
		}
		remove_variable = prime_minister
		random_scope_character = {
			limit = {
				NOT = {
					ruler = yes
					heir = yes
				}
				interest_group = {
					is_in_government = yes
				}
			}
			set_variable = prime_minister_trigger
			owner = {
				set_variable = {
					name = prime_minister
					value = PREV
				}
			}	
		}
	}
	ai_chance = {
		base = 0
	}
}
